# wasp-pig-latin
Pig Latin generator for WASP SW course

# Compile PigLatin:
ant build

# Run PigLatin:
java -jar build/jar/PigLatin.jar

# Run PigLatin:
java -jar build/jar/PigLatin.jar quiz

# Compile & Run PigLatin:
ant run

# Compile & Run PigLatin Quiz:
ant run -Dmode=quiz

# Compile and run unit tests
ant test

# Clean
ant clean

#Dependencies
Java SE JDK 8

Ant
